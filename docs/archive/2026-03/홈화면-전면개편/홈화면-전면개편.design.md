## 6. 신규 Hook: useWeather

```typescript
// src/lib/useWeather.ts
export function useWeather() {
  const [data, setData] = useState<WeatherData | null>(null)
  const [loading, setLoading] = useState(false)
  const [error, setError] = useState<string | null>(null)

  useEffect(() => {
    if (!navigator.geolocation) { setError('위치 미지원'); return }
    setLoading(true)
    navigator.geolocation.getCurrentPosition(
      async ({ coords }) => {
        const { latitude, longitude } = coords
        const apiKey = process.env.NEXT_PUBLIC_OPENWEATHER_API_KEY
        const url = `https://api.openweathermap.org/data/2.5/weather?lat=${latitude}&lon=${longitude}&units=metric&lang=kr&appid=${apiKey}`
        const res = await fetch(url)
        const json = await res.json()
        setData({
          city: json.name,
          temp: Math.round(json.main.temp),
          feels_like: Math.round(json.main.feels_like),
          description: json.weather[0].description,
          icon: json.weather[0].icon,
          humidity: json.main.humidity,
          wind_speed: json.wind.speed,
        })
        setLoading(false)
      },
      () => { setError('위치 권한이 필요해요'); setLoading(false) }
    )
  }, [])

  return { data, loading, error }
}
```

## 7. AI 채팅 전송 로직 (page.tsx)

```typescript
const handleSend = async () => {
  const text = inputText.trim()
  if (!text || isSending) return

  // 1. 사용자 메시지 추가
  const userMsg: ChatMessage = { id: Date.now().toString(), role: 'user', content: text, timestamp: new Date() }
  setMessages(prev => [...prev, userMsg])
  setInputText('')
  setIsSending(true)

  try {
    // 2. /api/chat 호출 (기존 mode:'save')
    const res = await fetch('/api/chat', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ mode: 'save', messages: [{ role: 'user', content: text }] }),
    })
    const data = await res.json()

    // 3. 저장 처리 (기존 로직)
    if (data.type === 'diary') await saveDiary({ ... })
    else if (data.type === 'moment') await addMoment({ ... })
    else if (data.type === 'idea') await addIdea({ ... })

    // 4. AI 응답 메시지 추가
    const aiMsg: ChatMessage = {
      id: (Date.now() + 1).toString(),
      role: 'ai',
      content: data.reply ?? getDefaultReply(data.type),
      timestamp: new Date(),
    }
    setMessages(prev => [...prev, aiMsg])
  } catch {
    setMessages(prev => [...prev, {
      id: (Date.now() + 1).toString(),
      role: 'ai',
      content: '저장 중 오류가 발생했어요 😢',
      timestamp: new Date(),
    }])
  } finally {
    setIsSending(false)
  }
}

function getDefaultReply(type: string) {
  if (type === 'diary') return '일기로 저장됐어요! 📔'
  if (type === 'moment') return '메모로 저장됐어요! ⚡'
  if (type === 'idea') return '아이디어로 저장됐어요! 💡'
  return '기록됐어요!'
}
```
